SELECT WORDS.Id,
WORDS.Comment,
type.name as "Type",
WORDS.Level,
ENGLISH.TEXT as "English",
GERMAN.TEXT as "German",
FRENCH.TEXT as "French",
POLISH.TEXT as "Polish",
KOREAN.TEXT as "Korean",
SPANISH.TEXT as "Spanish",
ITALIAN.TEXT as "Italian",
LATIN.TEXT as "Latin",
RUSSIAN.TEXT as "Russian",
CROATIAN.TEXT as "Croatian",
CZECH.TEXT as "Czech"


FROM WORDS 

left JOIN ENGLISH ON ENGLISH.ID = WORDS.ID 
left JOIN GERMAN ON GERMAN.ID = WORDS.ID 
left JOIN POLISH ON POLISH.ID = WORDS.ID 
left JOIN FRENCH ON FRENCH.ID = WORDS.ID 
left JOIN KOREAN ON KOREAN.ID = WORDS.ID 
left JOIN ITALIAN on ITALIAN.ID = WORDS.ID
left JOIN SPANISH on SPANISH.ID = WORDS.ID
left JOIN LATIN on LATIN.ID = WORDS.ID
left JOIN RUSSIAN on RUSSIAN.ID = WORDS.ID
left JOIN CROATIAN on CROATIAN.ID = WORDS.ID
left JOIN CZECH on CZECH.ID = WORDS.ID

INNER JOIN TYPE ON words.typeid = type.id 



ORDER BY `ID` ASC
